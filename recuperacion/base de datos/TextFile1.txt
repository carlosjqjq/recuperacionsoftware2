-- Crear la base de datos
CREATE DATABASE RutasTuristicas;

-- Usar la base de datos
USE RutasTuristicas;

-- Crear la tabla de clientes
CREATE TABLE Clientes (
    Id INT PRIMARY KEY IDENTITY,
    Nombre NVARCHAR(100),
    Email NVARCHAR(100),
    Edad INT,
    Tipo NVARCHAR(50)
);
